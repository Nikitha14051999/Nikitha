Table1
select COUNTRY_REGION,ROUND((TOTALCASES/POPULATION),4) AS PERCENTAGE FROM (select COUNTRY_REGION ,POPULATION,TOTALCASES from WORLDOMETER1 WHERE POPULATION IS NOT NULL) WHERE ROWNUM <= 5 ORDER BY PERCENTAGE DESC;

Table2
select COUNTRY_REGION,ROUND((ACTIVECASES/POPULATION),4) AS PERCENTAGE FROM (select COUNTRY_REGION ,POPULATION,ACTIVECASES from WORLDOMETER1 WHERE POPULATION IS NOT NULL) WHERE ROWNUM <= 5 ORDER BY PERCENTAGE DESC;

Table3
SELECT WHO_REGION, SUM(TOTALDEATHS) AS NUMOFDEATHS
FROM WORLDOMETER1
WHERE ROWNUM <= 5
GROUP BY WHO_REGION
ORDER BY NUMOFDEATHS DESC;
